Правила написания

  • В теоретических вопросах можно писать словами, схемами и псевдокодом.
  • В практических задачах нужен код на C.
  • В C-коде лучше объявлять переменные в начале блока, а не в заголовке for.
  • Для динамической памяти нужны проверки malloc/calloc == NULL и освобождение памяти.
  • Структуры данных в задачах лучше писать, даже если просят только одну процедуру.
  • Если просят одну операцию, не нужно писать все остальные операции, кроме необходимых вспомогательных.
  • Симметричные случаи можно не расписывать полностью, если явно указать, что они симметричны.

Термины

  • Двоичное B-дерево — двоичное дерево, в котором нет вершин степени 1.
  • B-дерево — страничное B-tree, где узел содержит несколько ключей.
  • КЛП — прямой обход: корень, левое поддерево, правое поддерево.
  • ЛКП — симметричный обход.
  • ЛПК — обратный обход, он же постфиксная запись для дерева выражения.

C-шпаргалка

Минимальный набор подключений

В большинстве экзаменационных задач хватит этих четырёх библиотек.

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <limits.h>
  • stdio.h — ввод, вывод, файлы.
  • stdlib.h — динамическая память, abs, завершение программы.
  • string.h — строки, строковые ключи, копирование памяти.
  • limits.h — INT_MAX для графов и бесконечности.

stdio.h — ввод/вывод

printf("%d\n", x);
scanf("%d", &x);

FILE *f;
f = fopen("input.txt", "r");
if (f == NULL) {
    return 1;
}
fscanf(f, "%d", &n);
fclose(f);
  • printf, scanf — стандартный ввод/вывод.
  • fprintf, fscanf — ввод/вывод через файл.
  • fopen, fclose — открыть и закрыть файл.
  • fgets, puts — работа со строками.

stdlib.h — память

Node *p;
p = (Node *)malloc(sizeof(Node));
if (p == NULL) {
    return NULL;
}

int *a;
a = (int *)calloc(n, sizeof(int));
if (a == NULL) {
    return 1;
}

free(a);
a = NULL;
  • malloc(size) — выделить блок памяти.
  • calloc(count, size) — выделить массив и заполнить нулями.
  • realloc(ptr, size) — изменить размер блока; лучше через временный указатель.
  • free(ptr) — освободить память.
  • abs(x), atoi(s), atof(s) — вспомогательные функции.

string.h — строки и ключи

int cmp;
cmp = strcmp(key, root->key);

if (cmp < 0) {
    /* key меньше */
} else if (cmp > 0) {
    /* key больше */
} else {
    /* нашли */
}

strncpy(node->key, key, KEY_SIZE - 1);
node->key[KEY_SIZE - 1] = '\0';
  • strcmp(a, b) — сравнить строки; для ключей использовать его, а не ==.
  • strlen(s) — длина строки.
  • strcpy, strncpy — копирование строки.
  • strchr(s, ch) — найти символ в строке.
  • memset, memcpy — заполнение и копирование памяти.

limits.h — бесконечность для графов

#define INF (INT_MAX / 4)

for (i = 0; i < n; i++) {
    dist[i] = INF;
}
dist[start] = 0;

if (dist[u] != INF && dist[u] + w < dist[v]) {
    dist[v] = dist[u] + w;
    parent[v] = u;
}
  • INT_MAX — максимум для int.
  • LONG_MAX — максимум для long.
  • Для Дейкстры лучше брать INT_MAX / 4, чтобы снизить риск переполнения при dist[u] + w.

ctype.h и math.h — по ситуации

#include <ctype.h>
#include <math.h>

if (isdigit((unsigned char)c)) { ... }
if (isspace((unsigned char)c)) { ... }

if (fabs(a - b) < 1e-9) { ... }
  • isdigit, isalpha, isspace — полезны для разбора выражений.
  • fabs — модуль double.
  • sqrt, pow — редко, но могут понадобиться в вычислениях.

Шаблон узла дерева

typedef struct Node {
    int value;
    struct Node *left;
    struct Node *right;
} Node;

Node *create_node(int value)
{
    Node *node;

    node = (Node *)malloc(sizeof(Node));
    if (node == NULL) {
        return NULL;
    }

    node->value = value;
    node->left = NULL;
    node->right = NULL;

    return node;
}

Шаблон ребра графа

typedef struct Edge {
    int from;
    int to;
    int weight;
} Edge;

Edge *edges;
edges = (Edge *)malloc(m * sizeof(Edge));
if (edges == NULL) {
    return 1;
}

Этот шаблон подходит для задач, где граф задан списком рёбер.

Шаблон стека

typedef struct Stack {
    int *data;
    int top;
    int capacity;
} Stack;

void init_stack(Stack *s, int capacity)
{
    s->data = (int *)malloc(capacity * sizeof(int));
    s->top = -1;
    s->capacity = capacity;
}

Стек нужен для постфиксной записи, перевода инфиксной записи в постфиксную и нерекурсивных обходов.

Что помнить в старом C-стиле

  • Объявлять переменные заранее: int i;, а не for (int i = 0; ...).
  • Всегда проверять результат malloc и calloc.
  • При удалении узлов дерева обязательно делать free.
  • Для строковых ключей сравнивать через strcmp.
  • Для функций, которые не меняют данные, использовать const: int search(const Node *root, int key).
